```yaml
name: Generate Snake

on:
  schedule:
    - cron: "0 0 * * *"
  workflow_dispatch:

jobs:
  generate:
    runs-on: ubuntu-latest
    steps:
      - uses: Platane/snk@v3
        with:
          github_user_name: Shrikrushn07
          outputs: |
            dist/github-snake.svg
            dist/github-snake-dark.svg?palette=github-dark

        with:
          target_branch: output
          build_dir: dist
        env:
```
